CREATE TABLE tb_veiculo (ID  SERIAL NOT NULL, ANO INTEGER, DISPONIVEIS INTEGER, MARCA VARCHAR(255), MODELO VARCHAR(255), PLACA VARCHAR(255), VERSAO INTEGER, PRIMARY KEY (ID))
CREATE INDEX index_placa ON tb_veiculo (placa)
CREATE TABLE tb_aluguel (ID  SERIAL NOT NULL, DATAALUGUEL DATE, DATADEVOLUCAO DATE, VALORDIARIA FLOAT, cliente_id INTEGER, veiculo_id INTEGER, PRIMARY KEY (ID))
CREATE TABLE tb_cliente (ID  SERIAL NOT NULL, CPF VARCHAR(15) UNIQUE, ENDERECO VARCHAR(100), NASCIMENTO VARCHAR(255), NOME VARCHAR(255), PRIMARY KEY (ID))
CREATE INDEX index_nome ON tb_cliente (nome)
ALTER TABLE tb_aluguel ADD CONSTRAINT FK_tb_aluguel_cliente_id FOREIGN KEY (cliente_id) REFERENCES tb_cliente (ID)
ALTER TABLE tb_aluguel ADD CONSTRAINT FK_tb_aluguel_veiculo_id FOREIGN KEY (veiculo_id) REFERENCES tb_veiculo (ID)
